\begin{DoxyEnumerate}

\item call \mbox{\hyperlink{xml__parser_8h_ab005d415013fe00f32e8a0ab5a6089d3}{aws\+\_\+xml\+\_\+node\+\_\+traverse()}} on the node to descend into the node with a new callback and user\+\_\+data.

\end{DoxyEnumerate}

return true to continue the parsing operation.
