49// Comparing value in a constant time manner
50_INLINE_ uint32_t
51secure_cmp(IN const uint8_t *a, IN const uint8_t *b, IN const uint32_t size)
52{
53 volatile uint8_t res = 0;
54
55 for(uint32_t i = 0; i < size; ++i)
56 {
57 res |= (a[i] ^ b[i]);
58 }
59
60 return (0 == res);
61}

94// Return 1 if equal 0 otherwise
95_INLINE_ uint32_t
96secure_cmp32(IN const uint32_t v1, IN const uint32_t v2)
97{
98#if defined(__aarch64__)
99 uint32_t res;
100 __asm__ __volatile__("cmp %w1, %w2; \n "
101 "cset %w0, EQ; \n"
102 : "=r"(res)
103 : "r"(v1), "r"(v2)
104 :);
105 return res;
106#elif defined(__x86_64__) || defined(__i386__)
107 uint32_t res;
108 __asm__ __volatile__("xor %%edx, %%edx; \n"
109 "cmp %1, %2; \n "
110 "sete %%dl; \n"
111 "mov %%edx, %0; \n"
112 : "=r"(res)
113 : "r"(v1), "r"(v2)
114 : "rdx");
115 return res;
116#else
117 // Insecure comparison: The main purpose of secure_cmp32 is to avoid
118 // branches and thus to prevent potential side channel attacks. To do that
119 // we normally leverage some CPU special instructions such as "sete"
120 // (for __x86_64__) and "cset" (for __aarch64__). When dealing with general
121 // CPU architectures, the interpretation of the line below is left for the
122 // compiler, which may lead to an insecure branch.
123 return (v1 == v2 ? 1 : 0);
124#endif
125}

127// Return 0 if v1 < v2, (-1) otherwise
128_INLINE_ uint32_t
129secure_l32_mask(IN const uint32_t v1, IN const uint32_t v2)
130{
131#if defined(__aarch64__)
132 uint32_t res;
133 __asm__ __volatile__("cmp %w2, %w1; \n "
134 "cset %w0, HI; \n"
135 : "=r"(res)
136 : "r"(v1), "r"(v2)
137 :);
138 return (res - 1);
139#elif defined(__x86_64__) || defined(__i386__)
140 uint32_t res;
141 __asm__ __volatile__("xor %%edx, %%edx; \n"
142 "cmp %1, %2; \n "
143 "setl %%dl; \n"
144 "dec %%edx; \n"
145 "mov %%edx, %0; \n"
146
147 : "=r"(res)
148 : "r"(v2), "r"(v1)
149 : "rdx");
150
151 return res;
152#else
153 // If v1 >= v2 then the subtraction result is 0^32||(v1-v2)
154 // else it will be 1^32||(v2-v1+1). Subsequently, negating the upper
155 // 32 bits gives 0 if v1 < v2 and otherwise (-1).
156 return ~((uint32_t)(((uint64_t)v1 - (uint64_t)v2) >> 32));
157#endif
158}
