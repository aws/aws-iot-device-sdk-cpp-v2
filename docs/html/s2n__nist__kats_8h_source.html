33static inline int FindMarker(FILE *infile, const char *marker)
34{
35 int8_t line[MAX_MARKER_LEN];
36 uint32_t i, len;
37
38 len = (int)strlen(marker);
39 if ( len > (MAX_MARKER_LEN - 1) ) {
40 len = MAX_MARKER_LEN-1;
41 }
42
43 for ( i=0; i<len; i++ ) {
44 if ( (line[i] = fgetc(infile)) == EOF ) {
45 return -1;
46 }
47 }
48 line[len] = '\0';
49
50 while ( 1 ) {
51 if ( !strncmp((char*)line, marker, len) ) {
52 return 0;
53 }
54
55 for ( i=0; i<len-1; i++ ) {
56 line[i] = line[i+1];
57 }
58 if ( (line[len-1] = fgetc(infile)) == EOF ) {
59 return -1;
60 }
61 line[len] = '\0';
62 }
63}

69static inline int ReadHex(FILE *infile, uint8_t *buf, uint32_t len, const char *str)
70{
71 int ch;
72 int started = 0;
73 uint8_t ich;
74
75 if (0 == len)
76 {
77 buf[0] = 0x00;
78 return 0;
79 }

88 while ((ch = fgetc(infile)) != EOF)
89 {
90 if (!isxdigit(ch)) {
91 if (!started) {
92 if (ch == '\n')
93 break;
94 else
95 continue;
96 } else
97 break;
98 }
99 started = 1;
100
101 if ((ch >= '0') && (ch <= '9')) {
102 ich = ch - '0';
103 }
104 else if ((ch >= 'A') && (ch <= 'F')) {
105 ich = ch - 'A' + 10;
106 }
107 else if ((ch >= 'a') && (ch <= 'f')) {
108 ich = ch - 'a' + 10;
109 }
110 else
111 {
112 /* shouldn't ever get here */
113 ich = 0;
114 }
115
116 for (uint32_t i = 0; i < len - 1; i++)
117 {
118 buf[i] = (buf[i] << 4) | (buf[i + 1] >> 4);
119 }
120 buf[len - 1] = (buf[len - 1] << 4) | ich;
121 }
122
123 return 0;
