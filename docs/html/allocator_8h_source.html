86/*
87 * Attempts to adjust the size of the pointed-to memory buffer from oldsize to
88 * newsize. The pointer (*ptr) may be changed if the memory needs to be
89 * reallocated.
90 *
91 * If reallocation fails, *ptr is unchanged, and this method raises an
92 * AWS_ERROR_OOM error.
93 */
94AWS_COMMON_API
95int aws_mem_realloc(struct aws_allocator *allocator, void **ptr, size_t oldsize, size_t newsize);
96/*
97 * Maintainer note: The above function doesn't return the pointer (as with
98 * standard C realloc) as this pattern becomes error-prone when OOMs occur.
99 * In particular, we want to avoid losing the old pointer when an OOM condition
100 * occurs, so we prefer to take the old pointer as an in/out reference argument
101 * that we can leave unchanged on failure.
