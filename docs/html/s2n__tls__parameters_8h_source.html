205/* The maximum size of a TLS record is 16389 bytes. This is; 1 byte for content
206 * type, 2 bytes for the protocol version, 2 bytes for the length field,
207 * and then up to 2^14 for the encrypted+compressed payload data.
208 */
209#define S2N_TLS_RECORD_HEADER_LENGTH 5
210#define S2N_TLS_MAXIMUM_FRAGMENT_LENGTH 16384
211/* Maximum TLS record length allows for 2048 octets of compression expansion and padding */
212#define S2N_TLS_MAXIMUM_RECORD_LENGTH (S2N_TLS_MAXIMUM_FRAGMENT_LENGTH + S2N_TLS_RECORD_HEADER_LENGTH + 2048)
213#define S2N_TLS_MAX_FRAG_LEN_EXT_NONE 0
214
215/* TLS1.3 has a max fragment length of 2^14 + 1 byte for the content type */
216#define S2N_TLS13_MAXIMUM_FRAGMENT_LENGTH 16385
217/* Max encryption overhead is 255 for AEAD padding */
218#define S2N_TLS13_MAXIMUM_RECORD_LENGTH (S2N_TLS13_MAXIMUM_FRAGMENT_LENGTH + S2N_TLS_RECORD_HEADER_LENGTH + 255)
219
220/* The maximum size of an SSL2 message is 2^14 - 1, as neither of the first two
221 * bits in the length field are usable. Per;
222 * http://www-archive.mozilla.org/projects/security/pki/nss/ssl/draft02.html
223 * section 1.1
224 */
225#define S2N_SSL2_RECORD_HEADER_LENGTH 2
226#define S2N_SSL2_MAXIMUM_MESSAGE_LENGTH 16383
227#define S2N_SSL2_MAXIMUM_RECORD_LENGTH (S2N_SSL2_MAXIMUM_MESSAGE_LENGTH + S2N_SSL2_RECORD_HEADER_LENGTH)

229/* s2n can use a "small" record length that is aligned to the dominant internet MTU;
230 * 1500 bytes, minus 20 bytes for an IP header, minus 20 bytes for a tcp
231 * header and 20 bytes for tcp/ip options (timestamp, sack etc) and a "large" record
232 * length that is designed to maximize throughput (fewer MACs per byte transferred
233 * and better efficiency of crypto engines).
234 */
235#define S2N_SMALL_RECORD_LENGTH (1500 - 20 - 20 - 20)
236#define S2N_SMALL_FRAGMENT_LENGTH (S2N_SMALL_RECORD_LENGTH - S2N_TLS_RECORD_HEADER_LENGTH)
