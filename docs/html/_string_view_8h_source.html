543 static const CharT *_s_search_substr(
544 const CharT *first1,
545 const CharT *last1,
546 const CharT *first2,
547 const CharT *last2)
548 {
549 const ptrdiff_t length2 = last2 - first2;
550 if (length2 == 0)
551 {
552 return first1;
553 }
554
555 ptrdiff_t length1 = last1 - first1;
556 if (length1 < length2)
557 {
558 return last1;
559 }
560
561 while (true)
562 {
563 length1 = last1 - first1;
564 if (length1 < length2)
565 {
566 return last1;
567 }
568
569 first1 = Traits::find(first1, length1 - length2 + 1, *first2);
570 if (first1 == 0)
571 {
572 return last1;
573 }
574
575 if (Traits::compare(first1, first2, length2) == 0)
576 {
577 return first1;
578 }
579
580 ++first1;
581 }
582 }
