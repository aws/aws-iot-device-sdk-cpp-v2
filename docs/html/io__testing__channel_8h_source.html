150static int s_testing_channel_handler_shutdown(
151 struct aws_channel_handler *handler,
152 struct aws_channel_slot *slot,
153 enum aws_channel_direction dir,
154 int error_code,
155 bool free_scarce_resources_immediately) {
156
157 struct testing_channel_handler *testing_handler = handler->impl;
158
159 /* If user has registered a callback, invoke it */
160 if (testing_handler->on_shutdown) {
161 testing_handler->on_shutdown(
162 dir, error_code, free_scarce_resources_immediately, testing_handler->on_shutdown_user_data);
163 }
164
165 if (dir == AWS_CHANNEL_DIR_WRITE) {
166 if (!slot->adj_left) {
167 /* Invoke the on_completion callbacks for any queued messages */
168 struct aws_linked_list_node *node = aws_linked_list_begin(&testing_handler->messages);
169 while (node != aws_linked_list_end(&testing_handler->messages)) {
170 struct aws_io_message *msg = AWS_CONTAINER_OF(node, struct aws_io_message, queueing_handle);
171
172 if (msg->on_completion) {
173 msg->on_completion(slot->channel, msg, AWS_IO_SOCKET_CLOSED, msg->user_data);
174 msg->on_completion = NULL;
175 }
176
177 node = aws_linked_list_next(node);
178 }
179 }
180 }
181
182 return aws_channel_slot_on_handler_shutdown_complete(slot, dir, error_code, free_scarce_resources_immediately);
183}

352static inline void testing_channel_drain_queued_tasks(struct testing_channel *testing) {
353 AWS_ASSERT(aws_channel_thread_is_callers_thread(testing->channel));
354
355 uint64_t now = 0;
356 uint64_t next_task_time = 0;
357 size_t count = 0;
358
359 while (true) {
360 aws_event_loop_current_clock_time(testing->loop, &now);
361 if (aws_task_scheduler_has_tasks(&testing->loop_impl->scheduler, &next_task_time) && (next_task_time <= now)) {
362 aws_task_scheduler_run_all(&testing->loop_impl->scheduler, now);
363 } else {
364 break;
365 }
366
367 /* NOTE: This will loop infinitely if there's a task the perpetually re-schedules another task.
368 * Consider capping the number of loops if we want to support that behavior. */
369 if ((++count % 1000) == 0) {
370 AWS_LOGF_WARN(
371 AWS_LS_IO_CHANNEL,
372 "id=%p: testing_channel_drain_queued_tasks() has looped %zu times.",
373 (void *)testing->channel,
374 count);
375 }
376 }

579static inline int testing_channel_send_data(
580 struct testing_channel *channel,
581 struct aws_byte_cursor data,
582 enum aws_channel_direction dir,
583 bool ignore_send_message_errors) {
584
585 struct aws_io_message *msg =
586 aws_channel_acquire_message_from_pool(channel->channel, AWS_IO_MESSAGE_APPLICATION_DATA, data.len);
587 ASSERT_NOT_NULL(msg);
588
589 ASSERT_TRUE(aws_byte_buf_write_from_whole_cursor(&msg->message_data, data));
590
591 int err;
592 if (dir == AWS_CHANNEL_DIR_READ) {
593 err = testing_channel_push_read_message(channel, msg);
594 } else {
595 err = testing_channel_push_write_message(channel, msg);
596 }
597
598 if (err) {
599 /* If an error happens, clean the message here. Else, the recipient of the message will take the ownership */
600 aws_mem_release(msg->allocator, msg);
601 }
602
603 if (!ignore_send_message_errors) {
604 ASSERT_SUCCESS(err);
605 }
606
607 return AWS_OP_SUCCESS;
608}
