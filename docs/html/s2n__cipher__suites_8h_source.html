49/* From RFC: https://tools.ietf.org/html/rfc8446#section-5.5
50 * For AES-GCM, up to 2^24.5 full-size records (about 24 million) may be
51 * encrypted on a given connection while keeping a safety margin of
52 * approximately 2^-57 for Authenticated Encryption (AE) security.
53 * S2N_TLS13_MAXIMUM_RECORD_NUMBER is 2^24.5 rounded down to the nearest whole number
54 * minus 1 for the key update message.
55 */
56#define S2N_TLS13_AES_GCM_MAXIMUM_RECORD_NUMBER ((uint64_t) 23726565)
