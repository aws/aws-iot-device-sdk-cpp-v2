81struct s2n_handshake_parameters {
82 /* Signature/hash algorithm pairs offered by the client in the signature_algorithms extension */
83 struct s2n_sig_scheme_list client_sig_hash_algs;
84
85 /* Signature/hash algorithm pairs offered by the server in the certificate request */
86 struct s2n_sig_scheme_list server_sig_hash_algs;
87
88 /* The cert chain we will send the peer. */
89 struct s2n_cert_chain_and_key *our_chain_and_key;
90
91 /* The subset of certificates that match the server_name presented in the ClientHello.
92 * In the case of multiple certificates matching a server_name, s2n will prefer certificates
93 * in FIFO order based on calls to s2n_config_add_cert_chain_and_key_to_store
94 *
95 * Note that in addition to domain matching, the key type for the certificate must also be
96 * suitable for a negotiation in order to be selected. The set of matching certs here are indexed
97 * by s2n_authentication_method.
98 *
99 * Example:
100 * - Assume certA is added to s2n_config via s2n_config_add_cert_chain_and_key_to_store
101 * - Next certB is added.
102 * - if certA matches www.foo.com and certB matches www.foo.com, s2n will prefer certA
103 *
104 * Note that in addition to domain matching, the key type for the certificate must also be
105 * suitable for a negotiation in order to be selected.
106 *
107 * Example:
108 * - Assume certA and certB match server_name www.foo.com
109 * - certA is ECDSA and certB is RSA.
110 * - Client only supports RSA ciphers
111 * - certB will be selected.
112 */
113 struct s2n_cert_chain_and_key *exact_sni_matches[S2N_CERT_TYPE_COUNT];
114 struct s2n_cert_chain_and_key *wc_sni_matches[S2N_CERT_TYPE_COUNT];
115 uint8_t exact_sni_match_exists;
116 uint8_t wc_sni_match_exists;
117};

148 /* Handshake type is a bitset, with the following
149 bit positions */
150 uint32_t handshake_type;
151
152/* Has the handshake been negotiated yet? */
153#define INITIAL 0x00
154#define NEGOTIATED 0x01
155#define IS_NEGOTIATED( type ) ( (type) & NEGOTIATED )
156
157/* Handshake is a full handshake */
158#define FULL_HANDSHAKE 0x02
159#define IS_FULL_HANDSHAKE( type ) ( (type) & FULL_HANDSHAKE )
160#define IS_RESUMPTION_HANDSHAKE( type ) ( !IS_FULL_HANDSHAKE( (type) ) && IS_NEGOTIATED ( (type) ) )
161
162/* Handshake uses perfect forward secrecy */
163#define TLS12_PERFECT_FORWARD_SECRECY 0x04
164
165/* Handshake needs OCSP status message */
166#define OCSP_STATUS 0x08
167#define IS_OCSP_STAPLED( type ) ( ( (type) & OCSP_STATUS ) != 0 )
168
169/* Handshake should request a Client Certificate */
170#define CLIENT_AUTH 0x10
171#define IS_CLIENT_AUTH_HANDSHAKE( type ) ( (type) & CLIENT_AUTH )
172
173/* Session Resumption via session-tickets */
174#define WITH_SESSION_TICKET 0x20
175#define IS_ISSUING_NEW_SESSION_TICKET( type ) ( (type) & WITH_SESSION_TICKET )
176
177/* Handshake requested a Client Certificate but did not get one */
178#define NO_CLIENT_CERT 0x40
179#define IS_CLIENT_AUTH_NO_CERT( type ) ( IS_CLIENT_AUTH_HANDSHAKE( (type) ) && ( (type) & NO_CLIENT_CERT) )
180
181/* A HelloRetryRequest was needed to proceed with the handshake */
182#define HELLO_RETRY_REQUEST 0x80
183
184/* Disguise a TLS1.3 handshake as a TLS1.2 handshake for backwards compatibility
185 * with some middleboxes: https://tools.ietf.org/html/rfc8446#appendix-D.4 */
186#define MIDDLEBOX_COMPAT 0x100
187#define IS_MIDDLEBOX_COMPAT_MODE( type ) ( (type) & MIDDLEBOX_COMPAT )
188
189 /* Which handshake message number are we processing */
190 int message_number;
191
192 /* State of the async pkey operation during handshake */
193 s2n_async_state async_state;
194
195 /* Indicates the CLIENT_HELLO message has been completely received */
196 unsigned client_hello_received:1;
197
198 /* Indicates the handshake blocked while trying to read or write data, and has been paused */
199 unsigned paused:1;
200
201 /* Set to 1 if the RSA verification failed */
202 unsigned rsa_failed:1;
203};
